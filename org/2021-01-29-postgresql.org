select
	t.id,
	t.industry_ids,
	t.report_type,
	t.create_time,
	t.modify_time,
	t.status,
	t.remark
from
	(
	select
		*,
		(
		select
			array_agg(industry_id) industry_ids
		from
			t_industry_report_relation tirr
		where
			industry_report_id = tir.id
		group by
			tir.id)
	from
		t_industry_report tir)t
where 1=ANY(industry_ids);
